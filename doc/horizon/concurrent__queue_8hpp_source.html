282  /*!
283  * @brief Swaps two concurrent_queues.
284  * @param left The left container.
285  * @param right The right container.
286  * @param leftLock The access token of the left queue.
287  * @param rightLock The access token of the right queue.

289  friend void swap(concurrent_queue& left, concurrent_queue& right, access_token const& leftLock, access_token const& rightLock) noexcept
290  {
291  swap(static_cast<concurrent_base&>(left), static_cast<concurrent_base&>(right), leftLock, rightLock);
292 
293  assert(leftLock.mutex()->native_handle() == left._containerAccess.native_handle());
294  assert(rightLock.mutex()->native_handle() == right._containerAccess.native_handle());
295 
296  // make sure both containers are not closed
297  // ThrowIfClosed();
298  // other.ThrowIfClosed();
299 
300  std::swap(left._container, right._container);
301 
302  left._containerCV.notify_all();
303  right._containerCV.notify_all();
304  }

306  /*!
307  * @brief Swaps two concurrent_queues.
308  *
309  * @note Calling this method takes ownership of both containers for the duration of the swap. If a container is currently owned by
310  * another thread, this method will block until the other thread releases the container. Both containers are locked simultaneously via
311  * defer_lock.
312  *
313  * @param left The left container.
314  * @param right The right container.
315  *
316  * @sa swap(concurrent_queue&, concurrent_queue&, access_token const&, access_token const&)
317  */
318  friend void swap(concurrent_queue& left, concurrent_queue& right) noexcept
319  {
320  // safety, do not swap (and more importantly, lock twice) the same object
321  if (&left == &right)return;
322 
323  auto leftLock = left.Guard(std::defer_lock);
324  auto rightLock = right.Guard(std::defer_lock);
325 
326  std::lock(leftLock, rightLock);
327 
328  swap(left, right, leftLock, rightLock);
329  }

372  /*!
373  * Waits until the container is not empty.
374  * Blocks the current thread execution until an element is in queue or the wait timeout is reached.
375  * Returns true if an element is in queue.
376  */
377  inline bool WaitForElementsInQueue(access_token& token, time_type const& waitTime) noexcept
378  {
379  // assert(waitTime.count() > 0);
380 
381  // early checks for either empty or closed.
382  // first check if we have elements in queue, then for closed
383  // this order allows emptying the queue after it has been closed
384  if (!empty(token)) return true;
385  if (is_closed()) return false;
386 
387  if (!_containerCV.wait_for(token, waitTime, [this, &token] { return !empty(token) || is_closed(); }))
388  return false; // timeout
389 
390  // ThrowIfClosed();
391  return CanModify();
392  }

203  /*!
204  * @brief Swaps two elements of the container if the container is owned by the calling thread.
205  * @param first The index of the first element.
206  * @param second The index of the second element.
207  * @param token The access token.
208  *
209  * @throws out_of_range If either first or second is out of bounds.
210  */
211  void SwapElements(access_type const& first, access_type const& second, access_token const& token)
212  {
213  // do not swap the same element
214  if (first == second) return;
215 
216  // <0 is (currently) impossible since access_type is unsigned
217  if (first >= size(token)) throw std::out_of_range("Parameter \"first\" is out of bounds.");
218  if (second >= size(token)) throw std::out_of_range("Parameter \"second\" is out of bounds.");
219 
220  CheckForOwnership(token);
221  std::iter_swap(begin() + first, begin() + second);
222  }

250  /*!
251  * @brief Gets an iterator to the beginning of the container.
252  *
253  * @details Iterators are THE problem for concurrent containers. There is no guarantee that the container is owned by the thread requesting
254  * the iterator, thus resulting in race conditions. I decided to expose the underlying container iterator, but only in "locked" methods. If
255  * you need to use iterators (and lets be honest, they are pretty neat), make sure you follow this principle:
256  * @code{.cpp}
257  * {
258  * auto token = container.Guard();
259  * auto it = container.begin(token);
260  * ...
261  * }
262  * @endcode
263  * This ensures that the container is locked until the iterator is out of scope (curly-brackets). It is not as nice as just using the
264  * standard begin, and prevents the "default" usage of containers in stl algorithms but emphasises the need for ownership.
265  * If I feel the need
266  * for it, I'll implement a custom iterator that locks the container until it is destroyed.
267  *
268  * @param token The access token.
269  * @warning This is inherently unsafe!
270  */
271  iterator begin(access_token const& token) noexcept
272  {
273  CheckForOwnership(token);
274  return _container.begin();
275  }

304  /*!
305  * @param style The separator style.
306  * @return Returns the separator symbol.
307  *
308  * @throws invalid_argument Style must not be Style::DETECT
309  */
310  static inline constexpr char Separator(Style const& style = Style::Native)
311  {
312  switch (style)
313  {
314  case Style::Windows:
315  return '\\';
316  case Style::Unix:
317  return '/';
318 
319  case Style::Detect:
320  throw std::invalid_argument("Detect is not a valid separator style.");
321 
322  default:
323  case Style::Native:
324  return Separator(
325  #if HORIZON_OS_WINDOWS
326  Style::Windows
327  #else
328  Style::Unix
329  #endif
330  );
331  }
332  }
333 
334  /*!
335  * @param style The path separator style.
336  * @return Returns the path separator symbol.
337  *
338  * @throws invalid_argument Style must not be Style::DETECT
339  */
340  static inline constexpr char PathSeparator(Style const& style = Style::Native)
341  {
342  switch (style)
343  {
344  case Style::Windows:
345  return ';';
346  case Style::Unix:
347  return ':';
348 
349  case Style::Detect:
350  throw std::invalid_argument("Detect is not a valid separator style.");
351 
352  default:
353  case Style::Native:
354  return PathSeparator(
355  #if HORIZON_OS_WINDOWS
356  Style::Windows
357  #else
358  Style::Unix
359  #endif
360  );
361  }
362  }

40  /*!
41  * Creates an asynchronous job.
42  *
43  * @tparam F The function template.
44  * @tparam Result Result type of the function.
45  * @param function The function to call. Note that no arguments are allowed. Use std::bind(...) if necessary.
46  * @param threadPool The treadpool to run the job on. If threadPool is null, the job will be lazy evaluated.
47  * If threadPool is set, the job is both lazy evaluated and will run on the specified threadpool, whichever comes first. (Default threadPool is
48  * the global threadPool).
49  *
50  * @return A job object.
51  */
52  template<class F,
53  class ... Args,
54  class Result = std::result_of_t<F&&(Args...)>>
55  std::shared_ptr<CALLABLE::Job<Result>>
56  async(F&& function, Args... args, std::shared_ptr<ThreadPool> const& threadPool = GetGlobalThreadPool()) noexcept
57  {
58  // create the job
59  auto job = CALLABLE::make_job(std::bind(std::forward<F>(function), std::forward<Args>(args)...));
60 
61  // schedule it on the thread pool if a pool is given
62  // TODO: what if the thread pool is closed? Currently, the job would still be lazy-evaluated (but this throws)...
63  if (threadPool)
64  threadPool->Schedule(job);
65 
66  // return the handle to the caller
67  // NOTE: do not use std::move here because of copy elision (https://stackoverflow.com/questions/19267408/why-does-stdmove-prevent-rvo)
68  return job;
69  }

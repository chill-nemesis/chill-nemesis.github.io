36  std::shared_ptr<RENDERAPI::RenderAPI> _api = nullptr;

44  [[nodiscard]] inline std::shared_ptr<RENDERAPI::RenderAPI> GetRenderAPI() const noexcept
45  { return _api; }
46 
47  /*!
48  * @brief Returns true if the calling thread is the current "owner" of the context.
49  */
50  [[nodiscard]] inline bool IsCurrentThreadOwning() const noexcept
51  { return std::this_thread::get_id() == _owningTheadID; }
52 
53  /*!
54  * @brief Returns true if no thread owns the context.
55  */
56  [[nodiscard]] inline bool HasNoOwningThread() const noexcept // runner up for method name: HasSock()
57  { return _owningTheadID == InvalidThread; }
58 
59  private:
60  /*!
61  * @brief Marks the thread id as invalid and informs the render api.
62  */
63  inline void RemoveOwner() noexcept // runner up for method name: GiveSock()
64  {
65  _owningTheadID = InvalidThread;
66  if (_api) _api->OnThreadRelease();
67  }
68 
69  /*!
70  * @brief Marks the calling thread as the new "owner" and informs the render api.
71  */
72  inline void MakeOwner() noexcept
73  {
74  _owningTheadID = std::this_thread::get_id();
75  if (_api) _api->OnNewThread();
76  }
77  };

19  /*!
20  * Provides a synchronisation mechanism for multiple objects.
21  * One object can be accessed unlimited and without synchronisation.
22  * The other object needs synchronisation access and can deny access to the first object while it is locked.
23  */
24  class CountingAccess
25  {
26  private:
27  std::atomic_uint _counter;
28  std::atomic_bool _locked;
29  std::mutex _lock;
30 
31  public:
32  /*!
33  * Increments the access counter.
34  * If lock has been called without release, this blocks the calling thread.
35  */
36  void Increment();
37 
38  /*!
39  * Decreases the access counter.
40  * Notifies the locking thread once counter is zero.
41  */
42  void Decrement();
43 
44  /*!
45  * Locks the mutex, blocks if access counter is not zero.
46  */
47  void Lock();
48 
49  /*!
50  * Unlokcs the mutex.
51  */
52  void Unlock();
53 
54  /*!
55  * Name forwarding for lock_guards
56  */
57  inline void lock()
58  { Lock(); }
59 
60  /*!
61  * Name forwarding for lock_guards
62  */
63  inline void unlock()
64  { Unlock(); }
65  };

26  enum AccessFrequency
27  {
28  /*!
29  * Seldom modified
30  * Seldom used

32  Stream = GL_STREAM_DRAW,

35  * Seldom modified
36  * Often used
37  */

40  /*!
41  * Often modified
42  * Often used
43  */
44  Dynamic = GL_DYNAMIC_DRAW
45  };

47  enum AccessUsage
48  {
49  /*!
50  * Data is modified by the application
51  * Data is used as a source for drawing or image specification
52  *
53  * CPU -> GPU
54  */
55  Draw = 0,
56 
57  /*!
58  * Data is modified by the graphics pipeline (e.g. compute shader)
59  * Data is queried/used by the application
60  *
61  * GPU -> CPU
62  */
63  Read = 1,
64 
65  /*!
66  * Data is modified by the graphi9cs pipeline (e.g. compute shader)
67  * Data is used as a source for drawing or image specification
68  *
69  * GPU -> GPU
70  */
71  Copy = 2
72  };
73 
74  private:
75  GLenum _usage;
76 
77  public:
78  constexpr explicit BufferDataUsage(AccessFrequency const& frequency = AccessFrequency::Static,
79  AccessUsage const& usage = AccessUsage::Draw) :
80  _usage(frequency + usage)
81  { }
82 
83  constexpr BufferDataUsage(GLenum const& usage = GL_STATIC_DRAW) :
84  _usage(usage)
85  {
86  assert(usage == GL_STATIC_DRAW ||
87  usage == GL_STATIC_READ ||
88  usage == GL_STATIC_COPY ||
89 
90  usage == GL_STREAM_DRAW ||
91  usage == GL_STREAM_READ ||
92  usage == GL_STREAM_COPY ||
93 
94  usage == GL_DYNAMIC_DRAW ||
95  usage == GL_DYNAMIC_READ ||
96  usage == GL_DYNAMIC_COPY);
97  }
98 
99  [[nodiscard]] inline GLenum const& GetUsage() const noexcept
100  { return _usage; }
101 
102  inline operator GLenum const&() const noexcept
103  { return _usage; }

224  /*!
225  * @brief Creates an empty buffer of maximumBufferSize.
226  *
227  * The buffer will be created on the context of the calling thread. If no context is registered for the calling thread,
228  * the global context is used instead.
229  *
230  * If the buffer already exists, its previous stored data is deleted.
231  * To update buffer data, use Update().
232  */
233  inline bool Create() noexcept override
234  { return Create(_maxBufferSize, nullptr, 0); }
235 
236  /*!
237  * @brief Creates an empty buffer of maximumBufferSize on the specified context.
238  *
239  * @param context The context of the object. If nullptr is passed, the buffer will be created on the context of the calling thread. If no
240  * context is registered for the calling thread, the global context is used instead.
241  */
242  inline bool Create(OpenGLContext context) noexcept override
243  { return Create(_maxBufferSize, nullptr, 0, true, std::move(context)); }
244 
245  /*!
246  * @brief Creates an empty buffer object of maxSize.
247  * This will update the maximum buffer size to match the new size.
248  *
249  * If the buffer already exists, its previous stored data is deleted.
250  * To update buffer data, use Update().
251  *
252  * If the given size exceeds MaximumBufferSize, the maximum buffer size is adjusted.
253  *
254  * @param maxSize The new maximum size of the buffer.
255  * @param context The context of the object. If nullptr is passed, the buffer will be created on the context of the calling thread. If no
256  * context is registered for the calling thread, the global context is used instead.
257  */
258  inline bool Create(size_t maxSize, OpenGLContext context = nullptr) noexcept
259  { return Create(maxSize, nullptr, 0, true, std::move(context)); }

261  /*!
262  * Creates a buffer from the given object.
263  *
264  * If the buffer already exists, its previous stored data is deleted.
265  * To update buffer data, use Update().
266  *
267  * @param bufferData The buffer data.
268  * @param adjustBufferSize If true, the maximum buffer size is adjusted to match the given data size.
269  * @param context The context of the object. If nullptr is passed, the buffer will be created on the context of the calling thread.
270  * If no context is registered for the calling thread, the global context is used instead.
271  */
272  template<typename T>
273  inline bool Create(T const& bufferData, bool adjustBufferSize = true, OpenGLContext context = nullptr) noexcept
274  { return Create(sizeof(T), static_cast<void const*>(&bufferData), 1, adjustBufferSize, std::move(context)); }

276  /*!
277  * Creates a buffer from the given data.

279  * If the buffer already exists, its previous stored data is deleted.
280  * To update buffer data, use Update().
281  *
282  * @param bufferData The buffer data.
283  * @param adjustBufferSize If true, the maximum buffer size is adjusted to match the given data size.

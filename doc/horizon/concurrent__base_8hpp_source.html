24  /*!
25  * @ingroup group_algorithm_concurrent
26  *
27  * @brief The base class for all concurrent containers.
28  * @details This class adds concurrent access utility and lock functionality. It acts as an interface to prevent accidental race conditions
29  * while working as a standard STL container. As with everything C/C++-related, this cannot guarantee or prevent malicious behaviour. However,
30  * this must be a conscious choice of the implementer to circumvent the safety-mechanics of the containers.
31  *
32  * You will find that all methods provide two signatures: A default (stl-like) implementation and one requiring an access_token. Access_tokens
33  * are supposed to mark ownership of the container. This is useful if you need to make multiple calls to the container without interference of
34  * other threads. An access_token can be obtained by using the Guard()-methods of the container. Once a token is out of scope (or manually
35  * unlocked) the ownership is released. Each "default" method signature will acquire a guard token and then call the guarded implementation.
36  */
37  class concurrent_base
38  {
39  public:
40  /*!
41  * @brief The access token type used to mark ownership
42  */
43  using access_token = std::unique_lock<std::mutex>;
44  /*!
45  * @brief Time type used for timeouts.
46  */
47  using time_type = std::chrono::nanoseconds;
48 
49  private:
50  std::atomic_bool _closeToken = false;
51 
52  protected:
53  /*!
54  * @brief The container mutex.
55  */
56  mutable std::mutex _containerAccess;
57 
58  /*!
59  * @brief The container condition variable used to signal changes in the container.
60  */
61  std::condition_variable_any _containerCV;
62 
63  // see https://stackoverflow.com/questions/27726818/stdcondition-variablewait-for-exits-immediately-when-given-stdchronodura
64  // (second answer) for the gcc work-around
65  // technically, this is expected behaviour. GCC and windows are just nice to the user and do not create an overflow
66  // the standard explicitly warn about this behaviour: "Note that rel_time must be small enough not to overflow when added to std::chrono::steady_clock::now()."
67  // (https://en.cppreference.com/w/cpp/thread/condition_variable/wait_for)
68  /*!
69  * @brief The default maximum wait time for containers.
70  */
71  static constexpr const time_type _maxWaitTime = std::chrono::hours(100);
72 
73  public:
74  /*
75  * @brief Locks the container and returns an access token for concurrent calls to the container
76  * Blocks if the lock is not available.
77  *
78  * @include <preprocessor/unique_lock.hpp>
79  */
80  AddLockGuardsMutable(_containerAccess);
81 
82 
83  /*!
84  * @return True if the container is closed (elements cannot be modified).
85  */
86  [[nodiscard]] inline bool is_closed() const
87  { return _closeToken; }
88 
89  /*!
90  * @brief Closes the container for modification and notifies all waiting threads.
91  */
92  void close()
93  {
94  //I need to lock access to closing while modifications might take place
95  auto token = Guard();
96 
97  // set the token to true. If it was previously true, do not notify
98  if (_closeToken.exchange(true)) return;
99 
100  // container was previously not closed, notify all waiting threads.
101  _containerCV.notify_all();
102  }
103 
104  /*!
105  * @brief (Re-) opens the queue.
106  */
107  inline void open() //Todo: enable overriding for containers where reopening is not possible?
108  { _closeToken = false; }
109 
110 
111  /*!
112  * @return Returns true if the container is empty.
113  */
114  inline bool empty() const
115  { return empty(Guard()); }
116 
117  /*!
118  * @details Checks if the container is empty.
119  * @param token The access token marking the ownership of the container.
120  * @return Returns true if the container is empty.
121  */
122  virtual bool empty(access_token const& token) const = 0;
123 
124  /*!
125  * @brief Friend declaration of swap. This requires both containers to be locked.
126  * @relates concurrent_base
127  * @param left The left container.
128  * @param right The right container.
129  * @param leftLock The lock of the left container.
130  * @param rightLock The lock of the right container.
131  *
132  * @note I strongly advice to use the following scheme (in the specialised swap methods for child containers):
133  *
134  * @code{.cpp}
135  * // safety, do not swap (and more importantly, lock twice) the same object
136  * if (&left == &right)return;
137  * auto leftLock = left.Guard(std::defer_lock);
138  * auto rightLock = right.Guard(std::defer_lock);
139  * std::lock(leftLock, rightLock);
140  *
141  * swap(left, right, leftLock, rightLock);
142  * @endcode
143  */
144  friend void swap(concurrent_base& left, concurrent_base& right, access_token const& leftLock, access_token const& rightLock) noexcept
145  {
146  left.CheckForOwnership(leftLock);
147  right.CheckForOwnership(rightLock);
148 
149  right._closeToken = left._closeToken.exchange(right._closeToken);
150  }
151 
152  /*!
153  * @brief Default swap operation for concurrent_base is not supported. Use @sa swap(concurrent_base&, concurrent_base&, access_token
154  * const&, access_token const&) instead.
155  * @relates concurrent_base
156  * @param left
157  * @param right
158  */
159  friend void swap(concurrent_base& left, concurrent_base& right) = delete;
160 
161 
162  protected:
163  /*!
164  * @return True if the container can be modified (close token is not set).
165  */
166  [[nodiscard]] inline bool CanModify() const
167  { return !(is_closed()); }
168 
169 
170  /*!
171  * @brief Checks if the given token belongs to this class.
172  * @param token The token to check.
173  *
174  * @note This has no effect in no-debug builds.
175  */
176  inline void CheckForOwnership(access_token const& token) const
177  { assert(token.mutex()->native_handle() == _containerAccess.native_handle()); }
178  };

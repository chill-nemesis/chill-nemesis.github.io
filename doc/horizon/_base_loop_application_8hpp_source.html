25  /*!
26  * Runs a main loop (inside the main thread).
27  * Callbacks are to be kept as fast as possible and must not block.
28  * There is no guarantee of the execution order of callbacks.
29  */
30  template<class L,
31  std::enable_if_t<std::is_base_of<PARALLEL::LOOP::Loop, L>::value, int> = 0>
32  class BaseLoopApplication : public IApplication
33  {
34  public:
35  using MainThreadCallback = std::function<void(void)>;
36  using LoopType = L;
37  using DeltaTime = typename LoopType::DeltaTime;
38 
39  private:
40  std::shared_ptr<CALLABLE::CallbackManager<void(void)>> _mainThreadCallbacks;
41  LoopType _loop;
42 
43 
44  public:
45  BaseLoopApplication() :
46  _mainThreadCallbacks(CALLABLE::CallbackManager<void(void)>::Create()),
47  // first is loop terminate predicate, second is the loop callback for each tick
48  _loop([] { return true; }, [this](DeltaTime) { _mainThreadCallbacks->Call(); })
49  { }
50 
51  ~BaseLoopApplication()
52  { Terminate(); }
53 
54  [[maybe_unused]] inline LoopType const& GetLoop() const
55  { return _loop; }
56 
57  /*!
58  * Terminates the running application.
59  */
60  inline void Terminate()
61  { _loop.Terminate(); }
62 
63  /*!
64  * Registers a callback for the main loop.
65  * Each callback gets called once per loop cycle, but there is no guarantee on order.
66  */
67  inline CALLABLE::CallbackHandle Register(MainThreadCallback&& callback)
68  { return std::move(_mainThreadCallbacks->Register(std::forward<MainThreadCallback>(callback))); }
69 
70  /*!
71  * Starts user threads, then runs the main thread in a loop.
72  */
73  inline void Run() final
74  {
75  StartThreads();
76  _loop.Run();
77  StopThreads();
78  }
79 
80 
81  protected:
82  /*!
83  * Enables the inherited class to start threads (from the main thread), before it is blocked inside the application loop.
84  */
85  virtual void StartThreads()
86  { }
87 
88  /*!
89  * Enables the inherited class to stop/join threads with the main thread after Terminate has been called.
90  */
91  virtual void StopThreads()
92  { }
93 
94  };
